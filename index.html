let fullHistory = []; let filteredHistory = []; let databaseData = {}; // --- CONFIGURATION --- // Items before this date are hidden by default const CUTOFF_DATE = new Date("2026-01-21T00:00:00Z"); function toLocalTime(dateStr) { try { const year = new Date().getFullYear(); return new Date(`${dateStr}, ${year} UTC`); } catch (e) { return null; } } function resolveName(data) { if (!data) return "Unknown"; if (typeof data === 'string') return data; if (typeof data === 'object') { if (data.name && typeof data.name === 'string') return data.name; return Object.values(data)[0] || "Unknown"; } return "Unknown"; } async function fetchData() { try { const t = Date.now(); const [hReq, dReq] = await Promise.all([ fetch(`./history.json?t=${t}`), fetch(`./database.json?t=${t}`) ]); fullHistory = await hReq.json(); databaseData = await dReq.json(); // Initial Filter filteredHistory = fullHistory.filter(entry => { const d = toLocalTime(entry.date); return d && d >= CUTOFF_DATE; }); if (fullHistory.length > filteredHistory.length) { document.getElementById('load-all-btn').classList.remove('hidden'); } updateDashboard(); renderItems(filteredHistory); } catch (err) { document.getElementById('feed').innerHTML = `
Connection Lost
`; } } function updateDashboard() { const reviewers = Object.keys(databaseData).filter(k => k !== 'last_sync'); // "New Hits" counts items post-cutoff document.getElementById('stat-total').innerText = filteredHistory.length; // "Total DB" counts everything in history.json document.getElementById('stat-db-total').innerText = fullHistory.length.toLocaleString(); document.getElementById('stat-sources').innerText = reviewers.length; document.getElementById('stat-time').innerText = new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit', hour12: false}); } function renderItems(data) { const container = document.getElementById('feed'); if (!data || data.length === 0) { container.innerHTML = `
Monitoring for new measurements...
`; return; } let lastHourLabel = ""; let htmlContent = ""; data.forEach(entry => { const model = resolveName(entry.item); const dateObj = toLocalTime(entry.date); const localStr = dateObj.toLocaleTimeString([], { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit', hour12: false }); const currentHourLabel = localStr.split(':')[0]; if (currentHourLabel !== lastHourLabel) { htmlContent += `
${currentHourLabel}:00 Block
`; lastHourLabel = currentHourLabel; } htmlContent += `
${entry.reviewer} ${localStr}

${model}

`; }); container.innerHTML = htmlContent; } // --- EVENTS --- document.getElementById('load-all-btn').addEventListener('click', () => { renderItems(fullHistory); document.getElementById('load-all-btn').classList.add('hidden'); // Update "New Hits" to match full history so stats don't look broken document.getElementById('stat-total').innerText = fullHistory.length.toLocaleString(); }); document.getElementById('search').addEventListener('input', (e) => { const val = e.target.value.toLowerCase(); // Search will look through whatever is currently rendered const source = (fullHistory.length === filteredHistory.length) ? fullHistory : filteredHistory; const filtered = source.filter(h => resolveName(h.item).toLowerCase().includes(val) || h.reviewer.toLowerCase().includes(val) ); renderItems(filtered); }); fetchData();
